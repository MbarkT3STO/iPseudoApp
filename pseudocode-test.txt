# Functions

    return result
endfunction

print sum

# Prime number checker function
function isPrime(n)
    if n <= 1 then
        return false
    endif
    for i = 2 to n - 1
        if n % i = 0 then
            return false
        endif
    endfor
    return true
endfunction

# Test the prime function
var number = 17
if isPrime(number) then
    print "The number is prime"
else
    print "The number is not prime"
endif

# Complex example: Fibonacci sequence
function fibonacci(n)
    if n <= 1 then
        return n
    endif
    var a = 0
    var b = 1
    for i = 2 to n
        var temp = a + b
        a = b
        b = temp
    endfor
    return b
endfunction

print "Fibonacci sequence:"
for i = 0 to 10
    var fib = fibonacci(i)
    print fib
endfor
